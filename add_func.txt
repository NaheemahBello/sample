#include<stdio.h>
 
int sum(int num1, int num2) {
   int num3;
   num3 = num1 + num2;
   return (num3);
}


int main() {
   int num1, num2, result;
 
   printf("\nEnter the two numbers  \n");
   scanf("%d %d", &num1, &num2);
 
   //Call Function Sum With Two Parameters
   result = sum(num1, num2);
 
   printf("The sum of %d + %d = %d ", num1, num2, result);
   return (0);
}
